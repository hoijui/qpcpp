51 void QK_sched_(uint8_t p) {
52 
53 #ifdef Q_USE_NAMESPACE
54  using namespace QP;
55 #endif
56 
57  uint8_t pin = QK_currPrio_; // save the initial priority
58  QActive *a;
59 #ifdef QK_TLS // thread-local storage used?
60  uint8_t pprev = pin;
61 #endif
62  do {
63  a = QF::active_[p]; // obtain the pointer to the AO
64  QK_currPrio_ = p; // this becomes the current task priority
65 
66 #ifdef QK_TLS // thread-local storage used?
67  if (p != pprev) { // are we changing threads?
68  QK_TLS(a); // switch new thread-local storage
69  pprev = p;
70  }
71 #endif
72  QS_BEGIN_NOCRIT_(QS_QK_SCHEDULE, QS::aoObj_, a)
73  QS_TIME_(); // timestamp
74  QS_U8_(p); // the priority of the active object
75  QS_U8_(pin); // the preempted priority
76  QS_END_NOCRIT_()
77 
78  QF_INT_ENABLE(); // unconditionally enable interrupts
79 
80  QEvent const *e = a->get_(); // get the next event for this AO
81  a->dispatch(e); // dispatch e to this AO
82  QF::gc(e); // garbage collect the event, if necessary
83 
84  // determine the next highest-priority AO ready to run
85  QF_INT_DISABLE(); // disable interrupts
86  p = QK_readySet_.findMax();
87 
88 #ifdef QK_NO_MUTEX
89  } while (p > pin); // is the new priority higher than initial?
90 #else // QK priority-ceiling mutexes allowed
91  } while ((p > pin) && (p > QK_ceilingPrio_));
92 #endif
93 
94  QK_currPrio_ = pin; // restore the initial priority
95 
96 #ifdef QK_TLS // thread-local storage used?
97  if (pin != (uint8_t)0) { // no extended context for the idle loop
98  a = QF::active_[pin]; // the pointer to the preempted AO
99  QK_TLS(a); // restore the original TLS
100  }
101 #endif
102 }
