103  }
104  else {
105  // (d) check source->super==target
106  if (m_state == path[0]) {
107  QEP_EXIT_(s) // exit the source
108  }
109  else { // (e) check rest of source==target->super->super..
110  // and store the entry path along the way
111  //
112  iq = (int8_t)0; // indicate that LCA not found
113  ip = (int8_t)1; // enter target and its superstate
114  path[1] = t; // save the superstate of target
115  t = m_state; // save source->super
116  // find target->super->super
117  r = QEP_TRIG_(path[1], QEP_EMPTY_SIG_);
118  while (r == Q_RET_SUPER) {
119  ++ip;
120  path[ip] = m_state; // store the entry path
121  if (m_state == s) { // is it the source?
122  iq = (int8_t)1; // indicate that LCA found
123  // entry path must not overflow
124  Q_ASSERT(ip < (int8_t)QEP_MAX_NEST_DEPTH_);
125  --ip; // do not enter the source
126  r = Q_RET_HANDLED; // terminate the loop
127  }
128  else { // it is not the source, keep going up
129  r = QEP_TRIG_(m_state, QEP_EMPTY_SIG_);
130  }
131  }
132  if (iq == (int8_t)0) { // the LCA not found yet?
133 
134  // entry path must not overflow
135  Q_ASSERT(ip < (int8_t)QEP_MAX_NEST_DEPTH_);
136 
137  QEP_EXIT_(s) // exit the source
138 
139  // (f) check the rest of source->super
140  // == target->super->super...
141  //
142  iq = ip;
143  r = Q_RET_IGNORED; // indicate LCA NOT found
144  do {
145  if (t == path[iq]) { // is this the LCA?
146  r = Q_RET_HANDLED; // indicate LCA found
147  ip = (int8_t)(iq - 1); // do not enter LCA
148  iq = (int8_t)(-1); // terminate the loop
149  }
150  else {
151  --iq; // try lower superstate of target
152  }
153  } while (iq >= (int8_t)0);
154 
155  if (r != Q_RET_HANDLED) { // LCA not found yet?
156  // (g) check each source->super->...
157  // for each target->super...
158  //
159  r = Q_RET_IGNORED; // keep looping
160  do {
161  // exit t unhandled?
162  if (QEP_TRIG_(t, Q_EXIT_SIG)
163  == Q_RET_HANDLED)
164  {
165  QS_BEGIN_(QS_QEP_STATE_EXIT,
166  QS::smObj_, this)
167  QS_OBJ_(this);
168  QS_FUN_(t);
169  QS_END_()
170

172  }
173  t = m_state; // set to super of t
174  iq = ip;
175  do {
176  if (t == path[iq]) { // is this LCA?
177  // do not enter LCA
178  ip = (int8_t)(iq - 1);
179  iq = (int8_t)(-1); //break inner
180  r = Q_RET_HANDLED; //break outer
181  }
182  else {

184  }

187  }
188  }
189  }
190  }
191  }
192  }
193  // retrace the entry path in reverse (desired) order...
194  for (; ip >= (int8_t)0; --ip) {
195  QEP_ENTER_(path[ip]) // enter path[ip]
196  }
197  t = path[0]; // stick the target into register
198  m_state = t; // update the current state
199 
200  // drill into the target hierarchy...
201  while (QEP_TRIG_(t, Q_INIT_SIG) == Q_RET_TRAN) {
202 
203  QS_BEGIN_(QS_QEP_STATE_INIT, QS::smObj_, this)
204  QS_OBJ_(this); // this state machine object
205  QS_FUN_(t); // the source (pseudo)state
206  QS_FUN_(m_state); // the target of the transition
207  QS_END_()
208 
209  ip = (int8_t)0;
210  path[0] = m_state;
211  (void)QEP_TRIG_(m_state, QEP_EMPTY_SIG_); // find superstate
212  while (m_state != t) {
213  ++ip;
214  path[ip] = m_state;
215  (void)QEP_TRIG_(m_state, QEP_EMPTY_SIG_); // find superstate
216  }
217  m_state = path[0];
218  // entry path must not overflow
219  Q_ASSERT(ip < (int8_t)QEP_MAX_NEST_DEPTH_);
220 
221  do { // retrace the entry path in reverse (correct) order...
222  QEP_ENTER_(path[ip]) // enter path[ip]
223  --ip;

226  t = path[0];
227  }

231  QS_SIG_(e->sig); // the signal of the event
232  QS_OBJ_(this); // this state machine object
233  QS_FUN_(src); // the source of the transition
234  QS_FUN_(t); // the new active state
235  QS_END_()
236 
237  }
238  else { // transition not taken
239 #ifdef Q_SPY

241

257  QS_END_()

259  }
