45 void QActive::postLIFO(QEvent const *e) {
46  QF_CRIT_STAT_
47  QF_CRIT_ENTRY_();
48 
49  QS_BEGIN_NOCRIT_(QS_QF_ACTIVE_POST_LIFO, QS::aoObj_, this)
50  QS_TIME_(); // timestamp
51  QS_SIG_(e->sig); // the signal of this event
52  QS_OBJ_(this); // this active object
53  QS_U8_(EVT_POOL_ID(e)); // the pool Id of the event
54  QS_U8_(EVT_REF_CTR(e)); // the ref count of the event
55  QS_EQC_(m_eQueue.m_nFree); // number of free entries
56  QS_EQC_(m_eQueue.m_nMin); // min number of free entries
57  QS_END_NOCRIT_()
58 
59  if (EVT_POOL_ID(e) != (uint8_t)0) { // is it a dynamic event?
60  EVT_INC_REF_CTR(e); // increment the reference counter
61  }
62 
63  if (m_eQueue.m_frontEvt == (QEvent *)0) { // is the queue empty?
64  m_eQueue.m_frontEvt = e; // deliver event directly
65  QACTIVE_EQUEUE_SIGNAL_(this); // signal the event queue
66  }
67  else { // queue is not empty, leave event in the ring-buffer
68  // queue must accept all posted events
69  Q_ASSERT(m_eQueue.m_nFree != (QEQueueCtr)0);
70 
71  ++m_eQueue.m_tail;
72  if (m_eQueue.m_tail == m_eQueue.m_end) { // need to wrap the tail?
73  m_eQueue.m_tail = (QEQueueCtr)0; // wrap around
74  }
75 
76  m_eQueue.m_ring[m_eQueue.m_tail] = m_eQueue.m_frontEvt;
77  m_eQueue.m_frontEvt = e; // put event to front
78 
79  --m_eQueue.m_nFree; // update number of free events
80  if (m_eQueue.m_nMin > m_eQueue.m_nFree) {
81  m_eQueue.m_nMin = m_eQueue.m_nFree; // update minimum so far
82  }
83  }
84  QF_CRIT_EXIT_();
85 }
