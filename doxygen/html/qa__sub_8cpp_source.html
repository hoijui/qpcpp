42 void QActive::subscribe(QSignal sig) const {
43  uint8_t p = m_prio;
44  Q_REQUIRE(((QSignal)Q_USER_SIG <= sig)
45  && (sig < QF_maxSignal_)
46  && ((uint8_t)0 < p) && (p <= (uint8_t)QF_MAX_ACTIVE)
47  && (QF::active_[p] == this));
48 
49  uint8_t i = Q_ROM_BYTE(QF_div8Lkup[p]);
50 
51  QF_CRIT_STAT_;
52  QF_CRIT_ENTRY_();
53 
54  QS_BEGIN_NOCRIT_(QS_QF_ACTIVE_SUBSCRIBE, QS::aoObj_, this)
55  QS_TIME_(); // timestamp
56  QS_SIG_(sig); // the signal of this event
57  QS_OBJ_(this); // this active object
58  QS_END_NOCRIT_()
59  // set the priority bit
60  QF_subscrList_[sig].m_bits[i] |= Q_ROM_BYTE(QF_pwr2Lkup[p]);
61  QF_CRIT_EXIT_();
62 }
