61 void QF::add_(QActive *a) {
62  uint8_t p = a->m_prio;
63 
64  Q_REQUIRE(((uint8_t)0 < p) && (p <= (uint8_t)QF_MAX_ACTIVE)
65  && (active_[p] == (QActive *)0));
66 
67  QF_CRIT_STAT_
68  QF_CRIT_ENTRY_();
69 
70  active_[p] = a; // registger the active object at this priority
71 
72  QS_BEGIN_NOCRIT_(QS_QF_ACTIVE_ADD, QS::aoObj_, a)
73  QS_TIME_(); // timestamp
74  QS_OBJ_(a); // the active object
75  QS_U8_(p); // the priority of the active object
76  QS_END_NOCRIT_()
77 
78  QF_CRIT_EXIT_();
79 }

83 
84  Q_REQUIRE(((uint8_t)0 < p) && (p <= (uint8_t)QF_MAX_ACTIVE)
