45 void QHsm::init(QEvent const *e) {
46  QStateHandler t;
47  QS_CRIT_STAT_
48 
49  // the top-most initial transition must be taken
50  Q_ALLEGE((*m_state)(this, e) == Q_RET_TRAN);
51 
52  t = (QStateHandler)&QHsm::top; // HSM starts in the top state
53  do { // drill into the target...
54  QStateHandler path[QEP_MAX_NEST_DEPTH_];
55  int8_t ip = (int8_t)0; // transition entry path index
56 
57 
58  QS_BEGIN_(QS_QEP_STATE_INIT, QS::smObj_, this)
59  QS_OBJ_(this); // this state machine object
60  QS_FUN_(t); // the source state
61  QS_FUN_(m_state); // the target of the initial transition
62  QS_END_()
63 
64  path[0] = m_state;
65  (void)QEP_TRIG_(m_state, QEP_EMPTY_SIG_);
66  while (m_state != t) {
67  ++ip;
68  path[ip] = m_state;
69  (void)QEP_TRIG_(m_state, QEP_EMPTY_SIG_);
70  }
71  m_state = path[0];
72  // entry path must not overflow
73  Q_ASSERT(ip < (int8_t)QEP_MAX_NEST_DEPTH_);
74 
75  do { // retrace the entry path in reverse (desired) order...
76  QEP_ENTER_(path[ip]); // enter path[ip]
77  --ip;
78  } while (ip >= (int8_t)0);
79 
80  t = path[0]; // current state becomes the new source
81  } while (QEP_TRIG_(t, Q_INIT_SIG) == Q_RET_TRAN);
82  m_state = t;
83 
84  QS_BEGIN_(QS_QEP_INIT_TRAN, QS::smObj_, this)
85  QS_TIME_(); // time stamp
86  QS_OBJ_(this); // this state machine object
87  QS_FUN_(m_state); // the new active state
88  QS_END_()
89 }
