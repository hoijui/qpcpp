70 void QF::run(void) {
71  QF::onStartup(); // startup callback
72 
73  for (;;) { // the bacground loop
74  QF_INT_DISABLE();
75  if (QF_readySet_.notEmpty()) {
76  uint8_t p = QF_readySet_.findMax();
77  QActive *a = active_[p];
78  QF_currPrio_ = p; // save the current priority
79  QF_INT_ENABLE();
80 
81  QEvent const *e = a->get_(); // get the next event for this AO
82  a->dispatch(e); // dispatch evt to the HSM
83  gc(e); // determine if event is garbage and collect it if so
84  }
85  else {
86  QF::onIdle(); // see NOTE01
87  }
88  }
89 }
