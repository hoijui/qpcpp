47 void QF::poolInit(void *poolSto, uint32_t poolSize, QEventSize evtSize) {
48  // cannot exceed the number of available memory pools
49  Q_REQUIRE(QF_maxPool_ < (uint8_t)Q_DIM(QF_pool_));
50  // please initialize event pools in ascending order of evtSize:
51  Q_REQUIRE((QF_maxPool_ == (uint8_t)0)
52  || (QF_EPOOL_EVENT_SIZE_(QF_pool_[QF_maxPool_ - 1]) < evtSize));
53  QF_EPOOL_INIT_(QF_pool_[QF_maxPool_], poolSto, poolSize, evtSize);
54  ++QF_maxPool_; // one more pool
55 }
