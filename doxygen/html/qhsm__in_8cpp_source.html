39 uint8_t QHsm::isIn(QStateHandler state) {
40  QStateHandler s = m_state;
41  QState r;
42  uint8_t inState = (uint8_t)0; // assume that this HSM is not in 'state'
43  do {
44  if (m_state == state) { // do the states match?
45  inState = (uint8_t)1; // match found, return TRUE
46  r = Q_RET_IGNORED; // break out of the loop
47  }
48  else {
49  r = QEP_TRIG_(m_state, QEP_EMPTY_SIG_);
50  }
51  } while (r != Q_RET_IGNORED); // QHsm_top state not reached
52  m_state = s; // restore the original state
53  return inState; // return the status
54 }
