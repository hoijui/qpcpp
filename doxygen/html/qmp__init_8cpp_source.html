42 void QMPool::init(void *poolSto, uint32_t poolSize, QMPoolSize blockSize) {
43  // The memory block must be valid
44  // and the poolSize must fit at least one free block
45  // and the blockSize must not be too close to the top of the dynamic range
46  Q_REQUIRE((poolSto != (void *)0)
47  && (poolSize >= (uint32_t)sizeof(QFreeBlock))
48  && ((QMPoolSize)(blockSize + (QMPoolSize)sizeof(QFreeBlock))
49  > blockSize));
50 
51  m_free = poolSto;
52 
53  // round up the blockSize to fit an integer number of pointers
54  m_blockSize = (QMPoolSize)sizeof(QFreeBlock); // start with just one
55  uint32_t nblocks = (uint32_t)1;// # free blocks that fit in a memory block
56  while (m_blockSize < blockSize) {
57  m_blockSize += (QMPoolSize)sizeof(QFreeBlock);
58  ++nblocks;
59  }
60  blockSize = m_blockSize; // use the rounded-up value from here on
61 
62  // the whole pool buffer must fit at least one rounded-up block
63  Q_ASSERT(poolSize >= (uint32_t)blockSize);
64 
65  // chain all blocks together in a free-list...
66  poolSize -= (uint32_t)blockSize; // don't chain the last block
67  m_nTot = (QMPoolCtr)1; // one (the last) block in the pool
68  QFreeBlock *fb = (QFreeBlock *)m_free;//start at the head of the free list
69  while (poolSize >= (uint32_t)blockSize) {
70  fb->m_next = &fb[nblocks]; // setup the next link
71  fb = fb->m_next; // advance to next block
72  poolSize -= (uint32_t)blockSize; // reduce the available pool size
73  ++m_nTot; // increment the number of blocks so far
74  }
75 
76  fb->m_next = (QFreeBlock *)0; // the last link points to NULL
77  m_nFree = m_nTot; // all blocks are free
78  m_nMin = m_nTot; // the minimum number of free blocks
79  m_start = poolSto; // the original start this pool buffer
80  m_end = fb; // the last block in this pool
81 
82  QS_CRIT_STAT_
83  QS_BEGIN_(QS_QF_MPOOL_INIT, QS::mpObj_, m_start)
84  QS_OBJ_(m_start); // the memory managed by this pool
85  QS_MPC_(m_nTot); // the total number of blocks
86  QS_END_()
87 }
