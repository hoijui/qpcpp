42 uint8_t QTimeEvt::rearm(QTimeEvtCtr nTicks) {
43  Q_REQUIRE((nTicks > (QTimeEvtCtr)0) /* cannot rearm with 0 ticks */
44  && (sig >= (QSignal)Q_USER_SIG) /* valid signal */
45  && (m_act != (QActive *)0)); // valid active object
46  uint8_t isArmed;
47  QF_CRIT_STAT_
48  QF_CRIT_ENTRY_();
49  m_ctr = nTicks;
50  if (m_prev == (QTimeEvt *)0) { // is this time event disarmed?
51  isArmed = (uint8_t)0;
52  m_next = QF_timeEvtListHead_;
53  if (QF_timeEvtListHead_ != (QTimeEvt *)0) {
54  QF_timeEvtListHead_->m_prev = this;
55  }
56  QF_timeEvtListHead_ = this;
57  m_prev = this; // mark the time evt in use
58  }
59  else { // the time event is armed
60  isArmed = (uint8_t)1;
61  }
62 
63  QS_BEGIN_NOCRIT_(QS_QF_TIMEEVT_REARM, QS::teObj_, this)
64  QS_TIME_(); // timestamp
65  QS_OBJ_(this); // this time event object
66  QS_OBJ_(m_act); // the active object
67  QS_TEC_(m_ctr); // the number of ticks
68  QS_TEC_(m_interval); // the interval
69  QS_U8_(isArmed); // was the timer armed?
70  QS_END_NOCRIT_()
71 
72  QF_CRIT_EXIT_();
73  return isArmed;
74 }
