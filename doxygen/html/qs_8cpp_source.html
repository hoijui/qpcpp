72 void QS::filterOn(uint8_t rec) {
73  if (rec == QS_ALL_RECORDS) {
74  uint8_t i;
75  for (i = (uint8_t)0; i < (uint8_t)sizeof(glbFilter_); ++i) {
76  glbFilter_[i] = (uint8_t)0xFF;
77  }
78  }
79  else {
80  glbFilter_[rec >> 3] |= (uint8_t)(1U << (rec & 0x07));
81  }
82 }

103 void QS::end(void) {
104  QS_INSERT_CHKSUM_BYTE()
105  QS_INSERT_BYTE(QS_FRAME)
106  if (QS_used_ > QS_end_) { // overrun over the old data?
107  QS_tail_ = QS_head_; // shift the tail to the old data
108  QS_used_ = QS_end_; // the whole buffer is used
109  }
110 }
