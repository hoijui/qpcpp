42 QEvent const *QEQueue::get(void) {
43  QEvent const *e;
44  QF_CRIT_STAT_
45 
46  QF_CRIT_ENTRY_();
47  if (m_frontEvt == (QEvent *)0) { // is the queue empty?
48  e = (QEvent *)0; // no event available at this time
49  }
50  else {
51  e = m_frontEvt;
52 
53  if (m_nFree != m_end) { // any events in the the ring buffer?
54  m_frontEvt = m_ring[m_tail]; // remove event from the tail
55  if (m_tail == (QEQueueCtr)0) { // need to wrap the tail?
56  m_tail = m_end; // wrap around
57  }
58  --m_tail;
59 
60  ++m_nFree; // one more free event in the ring buffer
61 
62  QS_BEGIN_NOCRIT_(QS_QF_EQUEUE_GET, QS::eqObj_, this)
63  QS_TIME_(); // timestamp
64  QS_SIG_(e->sig); // the signal of this event
65  QS_OBJ_(this); // this queue object
66  QS_U8_(EVT_POOL_ID(e)); // the pool Id of the event
67  QS_U8_(EVT_REF_CTR(e)); // the ref count of the event
68  QS_EQC_(m_nFree); // number of free entries
69  QS_END_NOCRIT_()
70  }
71  else {
72  m_frontEvt = (QEvent *)0; // the queue becomes empty
73 
74  QS_BEGIN_NOCRIT_(QS_QF_EQUEUE_GET_LAST, QS::eqObj_, this)
75  QS_TIME_(); // timestamp
76  QS_SIG_(e->sig); // the signal of this event
77  QS_OBJ_(this); // this queue object
78  QS_U8_(EVT_POOL_ID(e)); // the pool Id of the event
79  QS_U8_(EVT_REF_CTR(e)); // the ref count of the event
80  QS_END_NOCRIT_()
81  }
82  }
83  QF_CRIT_EXIT_();
84  return e;
85 }
